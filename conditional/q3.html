Program 3: Count Number of Digits

#include <stdio.h>
void main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    if(num >= 0 && num < 10)
        printf("1 digit");
    else if(num >= 10 && num < 100)
        printf("2 digits");
    else if(num >= 100 && num < 1000)
        printf("3 digits");
    else if(num >= 1000 && num < 10000)
        printf("4 digits");
    else if(num >= 10000 && num < 100000)
        printf("5 digits");
    else
        printf("More than 5 digits");
}
