Program 4: Grade System Based on Marks

#include <stdio.h>
void main() {
    int marks;
    printf("Enter marks: ");
    scanf("%d", &marks);
    if(marks >= 0 && marks < 40)
        printf("Fail");
    else if(marks >= 40 && marks < 60)
        printf("C Grade");
    else if(marks >= 60 && marks < 80)
        printf("B Grade");
    else if(marks >= 80 && marks <= 100)
        printf("A Grade");
    else
        printf("Invalid Marks");
}
