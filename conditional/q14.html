Program 14: Find Month Names by Number of Days

#include <stdio.h>
void main() {
    int days;
    printf("Enter number of days: ");
    scanf("%d", &days);
    if(days==30)
        printf("Months: April, June, September, November");
    else if(days==31)
        printf("Months: January, March, May, July, August, October, December");
    else if(days==28 || days==29)
        printf("Month: February");
    else
        printf("Invalid input");
}
