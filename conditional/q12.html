Program 12: Check if Four Characters are Equal

#include <stdio.h>
void main() {
    char a,b,c,d;
    printf("Enter 4 characters: ");
    scanf(" %c %c %c %c", &a,&b,&c,&d);
    if(a==b && b==c && c==d)
        printf("All characters are equal");
    else
        printf("Characters are not equal");
}
