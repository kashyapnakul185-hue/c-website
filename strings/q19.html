Q19 - Count Uppercase and Lowercase

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int i, upper=0, lower=0;
    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);
    for(i=0; str[i]!='\0'; i++){
        if(str[i]>='A' && str[i]<='Z') upper++;
        else if(str[i]>='a' && str[i]<='z') lower++;
    }
    printf("Uppercase: %d\nLowercase: %d", upper, lower);
    return 0;
}
