Q10 - Character at Index

#include <stdio.h>
#include <string.h>

int main() {
    char str[50];
    int index;
    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);
    printf("Enter index: ");
    scanf("%d", &index);
    if(index < 0) printf("Enter number >= 0 only");
    else if(index < strlen(str)-1) printf("Character: %c", str[index]);
    else printf("Out of bound");
    return 0;
}
