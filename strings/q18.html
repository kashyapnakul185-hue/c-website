Q18 - First Occurrence of Character

#include <stdio.h>
#include <string.h>

int main() {
    char str[100], ch;
    int i, flag=0;
    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);
    printf("Enter character to search: ");
    scanf("%c", &ch);
    for(i=0; str[i]!='\0'; i++){
        if(str[i]==ch){
            printf("Character found at index %d", i);
            flag=1;
            break;
        }
    }
    if(flag==0) printf("Character not found");
    return 0;
}
