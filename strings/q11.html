Q11 - Reverse String

#include <stdio.h>
#include <string.h>

int main() {
    char str[50];
    int i, len;
    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);
    len = strlen(str) - 1; // remove newline
    printf("Reversed string: ");
    for(i = len-1; i >= 0; i--) printf("%c", str[i]);
    return 0;
}
