Q3 - Convert String to Sentence Case

#include <stdio.h>
#include <string.h>

int main() {
    char str[50];
    int i;
    printf("Enter any sentence you like: ");
    fgets(str, sizeof(str), stdin);
    if(str[0] >= 'a' && str[0] <= 'z') str[0] -= 32;
    for(i = 1; str[i]; i++){
        if(str[i-1]=='.' && str[i] >= 'a' && str[i] <= 'z'){
            str[i] -= 32;
        }
    }
    printf("Sentence Case: %s", str);
    return 0;
}
