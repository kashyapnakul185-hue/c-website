Q1 - Convert String to Uppercase

#include <stdio.h>
#include <string.h>

int main() {
    char str[50];
    int i;
    printf("Enter any sentence you like: ");
    fgets(str, sizeof(str), stdin);
    for(i = 0; str[i]; i++){
        if(str[i] >= 'a' && str[i] <= 'z'){
            str[i] = str[i] - 32;
        }
    }
    printf("Uppercase version: %s", str);
    return 0;
}
