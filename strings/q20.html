Q20 - Replace Vowels

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int i;
    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);
    for(i=0; str[i]!='\0'; i++){
        if(strchr("AEIOUaeiou", str[i])) str[i]='*';
    }
    printf("String after replacement: %s", str);
    return 0;
}
