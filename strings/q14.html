Q14 - Word Count

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int i, words=1;
    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);
    for(i=0; str[i]!='\0'; i++){
        if(str[i]==' ' || str[i]=='\n' || str[i]=='\t') words++;
    }
    printf("Total words: %d", words);
    return 0;
}
