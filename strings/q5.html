Q5 - Toggle Case

#include <stdio.h>
#include <string.h>

int main() {
    char str[50];
    int i,l;
    printf("Enter any sentence you like: ");
    fgets(str, sizeof(str), stdin);
    l = strlen(str);
    for(i = 0; i < l; i++){
        if(str[i] >= 'A' && str[i] <= 'Z') str[i] += 32;
        else if(str[i] >= 'a' && str[i] <= 'z') str[i] -= 32;
    }
    printf("Toggled string: %s", str);
    return 0;
}
