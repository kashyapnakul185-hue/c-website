Q17 - Count Alphabets, Digits, Special Characters

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int i, alpha=0, digits=0, special=0;
    printf("Enter string: ");
    fgets(str, sizeof(str), stdin);
    for(i=0; str[i]!='\0'; i++){
        if((str[i]>='a' && str[i]<='z') || (str[i]>='A' && str[i]<='Z')) alpha++;
        else if(str[i]>='0' && str[i]<='9') digits++;
        else special++;
    }
    printf("Alphabets: %d\nDigits: %d\nSpecial Characters: %d", alpha, digits, special);
    return 0;
}
