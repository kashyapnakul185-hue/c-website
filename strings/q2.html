Q2 - Convert String to Lowercase

#include <stdio.h>
#include <string.h>

int main() {
    char str[50];
    int i;
    printf("Enter any sentence you like: ");
    fgets(str, sizeof(str), stdin);
    for(i = 0; str[i]; i++){
        if(str[i] >= 'A' && str[i] <= 'Z'){
            str[i] = str[i] + 32;
        }
    }
    printf("Lowercase version: %s", str);
    return 0;
}
