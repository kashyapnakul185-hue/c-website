Average Marks

// op11_average_marks.h
#include <stdio.h>
void main() {
    int n, i, marks, sum=0;
    printf("Enter number of students: ");
    scanf("%d", &n);
    for(i=1;i<=n;i++){
        printf("Enter marks of student %d: ", i);
        scanf("%d",&marks);
        sum += marks;
    }
    printf("Average marks: %.2f", (float)sum/n);
}
