Program 10 – Password Masking with *

#include <stdio.h>
#include <conio.h>

void main() {
    char password[20];
    int i = 0;
    char ch;

    printf("Enter password: ");

    while(1) {
        ch = getch();
        if(ch == 13) { // Enter key ASCII = 13
            break;
        }
        password[i] = ch;
        i++;
        printf("*");
    }
    password[i] = '\\0';
    printf("\\nYour password is: %s\\n", password);
}
