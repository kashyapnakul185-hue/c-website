Program 11 – ASCII Values of Two Characters

#include <stdio.h>

void main() {
    char ch1, ch2;

    printf("Enter first character: ");
    scanf(" %c", &ch1);

    printf("Enter second character: ");
    scanf(" %c", &ch2);

    printf("ASCII value of '%c' = %d\\n", ch1, ch1);
    printf("ASCII value of '%c' = %d\\n", ch2, ch2);
}
