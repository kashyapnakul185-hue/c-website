Q1. Binary of a Number

#include <stdio.h>
void main() {
    int num, rem, arr[16], x = 0, i;
    printf("Enter the number to find its binary: ");
    scanf("%d", &num);
    while (num != 0) {
        rem = num % 2;
        arr[x++] = rem;
        num /= 2;
    }
    printf("Binary: ");
    for (i = x - 1; i >= 0; i--) {
        printf("%d", arr[i]);
    }
    printf("\n");
}
