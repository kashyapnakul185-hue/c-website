Q2. Check Divisibility by Prime

#include <stdio.h>
void main() {
    int num, prime, got = 0;
    printf("Enter a number: ");
    scanf("%d", &num);
    for (int i = 2; i <= num / 2; i++) {
        prime = 1;
        for (int j = 2; j <= i / 2; j++) {
            if (i % j == 0) { prime = 0; break; }
        }
        if (prime && num % i == 0) { got = 1; break; }
    }
    if (got)
        printf("The number %d is divisible by a prime number\n", num);
    else
        printf("The number %d is not divisible by a prime number\n", num);
}
