Q3. Check Prime Number

#include <stdio.h>
void main() {
    int num, flag = 1;
    printf("Enter a number: ");
    scanf("%d", &num);
    if (num <= 1) flag = 0;
    for (int i = 2; i <= num / 2 && flag; i++) {
        if (num % i == 0) flag = 0;
    }
    if (flag)
        printf("%d is a Prime number\n", num);
    else
        printf("%d is not a Prime number\n", num);
}
