Program 5: Matrix Multiplication

#include <stdio.h>
void main() {
    int a[2][2], b[2][2], mul[2][2], i, j, k;
    printf("Enter 4 numbers for first matrix: ");
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            scanf("%d",&a[i][j]);
    printf("Enter 4 numbers for second matrix: ");
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            scanf("%d",&b[i][j]);
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            mul[i][j]=0;
            for(k=0;k<2;k++)
                mul[i][j] += a[i][k]*b[k][j];
        }
    }
    printf("Multiplication Matrix:\\n");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++)
            printf("%d\\t", mul[i][j]);
        printf("\\n");
    }
}
