Program 4: Matrix Subtraction

#include <stdio.h>
void main() {
    int a[2][2], b[2][2], sub[2][2], i, j;
    printf("Enter 4 numbers for first matrix: ");
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            scanf("%d",&a[i][j]);
    printf("Enter 4 numbers for second matrix: ");
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            scanf("%d",&b[i][j]);
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            sub[i][j] = a[i][j] - b[i][j];
    printf("Subtraction Matrix:\\n");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++)
            printf("%d\\t", sub[i][j]);
        printf("\\n");
    }
}
