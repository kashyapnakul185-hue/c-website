Program 1: Matrix Input

#include <stdio.h>
void main() {
    int mat[2][2], i, j;
    printf("Enter 4 numbers: ");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            scanf("%d", &mat[i][j]);
        }
    }
    printf("Matrix:\\n");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            printf("%d\\t", mat[i][j]);
        }
        printf("\\n");
    }
}
