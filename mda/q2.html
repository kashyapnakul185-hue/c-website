Program 2: Matrix Transpose

#include <stdio.h>
void main() {
    int mat[2][2], trans[2][2], i, j;
    printf("Enter 4 numbers for 2x2 matrix: ");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            scanf("%d",&mat[i][j]);
            trans[j][i] = mat[i][j];
        }
    }
    printf("Transpose Matrix:\\n");
    for(i=0;i<2;i++){
        for(j=0;j<2;j++){
            printf("%d\\t", trans[i][j]);
        }
        printf("\\n");
    }
}
