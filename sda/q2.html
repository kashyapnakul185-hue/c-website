Q2 - Print Array in Reverse

// Q2: Print array in reverse
#include <stdio.h>

void main() {
    int array[5], i;
    printf("Enter 5 array elements: ");
    for (i = 0; i < 5; i++) 
        scanf("%d", &array[i]);
    printf("Array in reverse: ");
    for (i = 4; i >= 0; i--) 
        printf("%d ", array[i]);
    printf("\n");
}
