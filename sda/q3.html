Q3 - Copy Array into Another Array

// Q3: Copy array into another array
#include <stdio.h>

void main() {
    int array1[5], array2[5], i;
    printf("Enter 5 elements for array1: ");
    for (i = 0; i < 5; i++) scanf("%d", &array1[i]);
    for (i = 0; i < 5; i++) array2[i] = array1[i];
    printf("Copied array2: ");
    for (i = 0; i < 5; i++) printf("%d ", array2[i]);
    printf("\n");
}
