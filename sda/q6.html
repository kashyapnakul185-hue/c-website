Q6 - Print Positive Numbers

// Q6: Print positive numbers
#include <stdio.h>

void main() {
    int array[5], i;
    printf("Enter 5 array elements: ");
    for (i = 0; i < 5; i++) scanf("%d", &array[i]);
    printf("Positive numbers: ");
    for (i = 0; i < 5; i++) if (array[i] > 0) printf("%d ", array[i]);
    printf("\n");
}
