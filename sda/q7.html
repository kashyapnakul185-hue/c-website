Q7 - Square All Values

// Q7: Square all values
#include <stdio.h>

void main() {
    int array[5], i;
    printf("Enter 5 array elements: ");
    for (i = 0; i < 5; i++) scanf("%d", &array[i]);
    for (i = 0; i < 5; i++) array[i] *= array[i];
    printf("Squared values: ");
    for (i = 0; i < 5; i++) printf("%d ", array[i]);
    printf("\n");
}
