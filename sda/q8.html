Q8 - Print Odd Numbers

// Q8: Print odd numbers
#include <stdio.h>

void main() {
    int array[5], i;
    printf("Enter 5 array elements: ");
    for (i = 0; i < 5; i++) scanf("%d", &array[i]);
    printf("Odd numbers: ");
    for (i = 0; i < 5; i++) if (array[i] % 2 != 0) printf("%d ", array[i]);
    printf("\n");
}
