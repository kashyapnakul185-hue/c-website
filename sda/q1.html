Q1 - Display Array

// Q1: Display array elements
#include <stdio.h>

void main() {
    int array[5], i;
    printf("Enter 5 array elements: ");
    for (i = 0; i < 5; i++) 
        scanf("%d", &array[i]);
    printf("Array elements: ");
    for (i = 0; i < 5; i++) 
        printf("%d ", array[i]);
    printf("\n");
}
