Q8 - Swap using Call by Function

#include <stdio.h>
#include <conio.h>

void swap_function8(int a, int b, int *x, int *y) {
int temp = a;
a = b;
b = temp;
*x = a;
*y = b;
}

void main() {
int a = 15, b = 25;
printf("Before swapping: a=%d, b=%d\\n", a, b);
swap_function8(a, b, &a, &b);
printf("After swapping: a=%d, b=%d\\n", a, b);
}
