Q4 - Local and Global Variable Programs

#include <stdio.h>
#include <math.h>

int global_num4;

void factorial4() {
int i, fact = 1;
for (i = 1; i <= global_num4; i++)
fact *= i;
printf("Factorial of %d = %d\\n", global_num4, fact);
}

void reverse4() {
int rev = 0, temp = global_num4;
while (temp != 0) {
rev = rev * 10 + temp % 10;
temp /= 10;
}
printf("Reverse of %d = %d\\n", global_num4, rev);
}

void palindrome4() {
int rev = 0, temp = global_num4;
while (temp != 0) {
rev = rev * 10 + temp % 10;
temp /= 10;
}
if (rev == global_num4)
printf("%d is a palindrome\\n", global_num4);
else
printf("%d is not a palindrome\\n", global_num4);
}

void prime4() {
int i, flag = 0;
for (i = 2; i <= global_num4 / 2; i++) {
if (global_num4 % i == 0) {
flag = 1;
break;
}
}
if (global_num4 <= 1)
printf("%d is not a prime number\\n", global_num4);
else if (flag == 0)
printf("%d is a prime number\\n", global_num4);
else
printf("%d is not a prime number\\n", global_num4);
}

void armstrong4() {
int temp = global_num4, sum = 0, rem, digits = 0;
while (temp != 0) {
digits++;
temp /= 10;
}
temp = global_num4;
while (temp != 0) {
rem = temp % 10;
sum += pow(rem, digits);
temp /= 10;
}
if (sum == global_num4)
printf("%d is an Armstrong number\\n", global_num4);
else
printf("%d is not an Armstrong number\\n", global_num4);
}

void fibonacci4() {
int i, t1 = 0, t2 = 1, next;
printf("Fibonacci series up to %d terms: ", global_num4);
for (i = 1; i <= global_num4; i++) {
printf("%d ", t1);
next = t1 + t2;
t1 = t2;
t2 = next;
}
printf("\\n");
}

void main() {
global_num4 = 5;
factorial4();
reverse4();
palindrome4();
prime4();
armstrong4();
fibonacci4();
}
