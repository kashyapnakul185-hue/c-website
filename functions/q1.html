Program 1: Multiple Operations Using Functions

#include <stdio.h>
#include <conio.h>

int input1() {
    int x;
    printf("Enter the number: ");
    scanf("%d", &x);
    return x;
}

void output1(int r) {
    printf("%d", r);
}

void add1() {
    int x, y, z;
    x = input1();
    y = input1();
    z = x + y;
    output1(z);
}

void sub1() {
    int x, y, z;
    x = input1();
    y = input1();
    z = x - y;
    output1(z);
}

void main() {
    clrscr();
    printf("Addition\\n");
    add1();
    printf("\\nSubtraction\\n");
    sub1();
    getch();
}
